## how to use log4cxx
LoggerPtr log = Logger::getLogger("NAME");

LOG4CXX_FATAL(log, "fatal output...");
LOG4CXX_ERROR(log, "error output...");
LOG4CXX_WARN(log, "warn output...");
LOG4CXX_INFO(log, "info output...");
LOG4CXX_DEBUG(log, "debug output...");
